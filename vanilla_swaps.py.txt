"""
vanilla_swaps.py  —  Vanilla USD Interest Rate Swap Pricer
===========================================================
  • 100 synthetic USD IRS priced with QuantLib + PySpark (parallel)
  • Single-curve SOFR framework: same curve for discounting & forwarding
  • Runs iteratively until NPV values converge  (max|ΔNPV| < $1.00)
  • Correct DV01  = +1bp parallel SOFR shift (single curve)
  • Correct Duration = DV01 / (notional × 0.0001)  [years]
  • Outputs: vanillaSwap.csv  +  vanillaSwapPortfolio.csv

# 1.  MARKET DATA  —  USD SOFR zero curve (2024-01-15)
# ══════════════════════════════════════════════════════════════════════════════

VALUATION_DATE = date(2024, 1, 15)

# Continuous ACT/365 zero rates — realistic 2024 SOFR inverted-curve levels
_SOFR_ZEROS = [
    #  tenor_y    zero_rate
    (  0.0833,    0.0533),   # 1M  — near Fed Funds effective rate
    (  0.2500,    0.0530),   # 3M
    (  0.5000,    0.0522),   # 6M
    (  1.0000,    0.0490),   # 1Y
    (  2.0000,    0.0455),   # 2Y
    (  3.0000,    0.0440),   # 3Y
    (  5.0000,    0.0428),   # 5Y
    (  7.0000,    0.0422),   # 7Y
    ( 10.0000,    0.0420),   # 10Y
    ( 15.0000,    0.0425),   # 15Y
    ( 20.0000,    0.0430),   # 20Y
    ( 30.0000,    0.0440),   # 30Y
]

# Approximate par rates by tenor — used to centre fixed-rate distribution
_PAR_APPROX = {
    1: 0.0490, 2: 0.0455, 3: 0.0440, 5: 0.0428,
    7: 0.0422, 10: 0.0420, 15: 0.0425, 20: 0.0430, 30: 0.0440,
}


def make_curve_df() -> pd.DataFrame:
    return pd.DataFrame(_SOFR_ZEROS, columns=["tenor_y", "zero_rate"])


def build_sofr_curve(val_date: ql.Date,
                     curve_df: pd.DataFrame) -> ql.YieldTermStructureHandle:
    """Linear zero curve — single SOFR curve for both discount and forward."""
    dc = ql.Actual365Fixed()
    dates = [val_date]
    rates = [float(curve_df["zero_rate"].iloc[0])]
    for _, row in curve_df.iterrows():
        months = max(1, int(round(float(row["tenor_y"]) * 12)))
        dates.append(val_date + ql.Period(months, ql.Months))
        rates.append(float(row["zero_rate"]))
    curve = ql.ZeroCurve(
        dates, rates, dc, ql.NullCalendar(),
        ql.Linear(), ql.Continuous, ql.Annual,
    )
    curve.enableExtrapolation()
    return ql.YieldTermStructureHandle(curve)

# 2.  TRADE DATA  —  100 synthetic spot-starting USD IRS
# ══════════════════════════════════════════════════════════════════════════════

def make_trade_data(n: int = 100) -> List[Dict[str, Any]]:
    """
    Generate n USD IRS trades.
      • Effective date = valuation + 2 days (spot)
      • Fixed rate = approx par ± spread  (centred around current SOFR levels)
      • USD conventions: SA fixed 30/360 BondBasis, quarterly float Act/360
    """
    import random
    random.seed(42)

    base      = VALUATION_DATE
    effective = base + timedelta(days=2)          # T+2 spot settlement

    tenors        = [1, 2, 3, 5, 7, 10, 15, 20, 30]
    notionals     = [1e6, 2e6, 5e6, 10e6, 20e6, 50e6]
    # Spreads vs approximate par — covers ATM, OTM, and ITM swaps
    spreads       = [-0.015, -0.010, -0.0075, -0.005, -0.0025,
                      0.000,  0.0025,  0.005,  0.0075,  0.010, 0.015, 0.020]
    books         = ["IRD-NY", "IRD-LDN", "IRD-ASIA", "MACRO"]
    counterparties = [f"CPTY-{c:02d}" for c in range(1, 11)]
    swap_types    = ["PAYER", "RECEIVER"]

    trades = []
    for i in range(n):
        tenor_y    = random.choice(tenors)
        notional   = float(random.choice(notionals))
        spread     = random.choice(spreads)
        swap_type  = random.choice(swap_types)
        book       = random.choice(books)
        cpty       = random.choice(counterparties)
        fixed_rate = round(max(0.005, _PAR_APPROX[tenor_y] + spread), 4)

        try:
            maturity = date(effective.year + tenor_y, effective.month, effective.day)
        except ValueError:
            maturity = date(effective.year + tenor_y, effective.month, effective.day - 1)

        trades.append({
            "trade_id":       f"IRS-{i + 1:04d}",
            "book":           book,
            "counterparty":   cpty,
            "option_type":    swap_type,
            "notional":       notional,
            "fixed_rate":     fixed_rate,
            "tenor_y":        tenor_y,
            "effective_date": effective,
            "maturity_date":  maturity,
            "valuation_date": base,
        })
    return trades


# ══════════════════════════════════════════════════════════════════════════════
# 3.  SCHEMAS
# ══════════════════════════════════════════════════════════════════════════════

TRADE_SCHEMA = StructType([
    StructField("trade_id",       StringType()),
    StructField("book",           StringType()),
    StructField("counterparty",   StringType()),
    StructField("option_type",    StringType()),
    StructField("notional",       DoubleType()),
    StructField("fixed_rate",     DoubleType()),
    StructField("tenor_y",        IntegerType()),
    StructField("effective_date", DateType()),
    StructField("maturity_date",  DateType()),
    StructField("valuation_date", DateType()),
])

RESULT_SCHEMA = StructType([
    StructField("fixed_npv",  DoubleType()),   # PV of fixed leg  (negative for payer)
    StructField("float_npv",  DoubleType()),   # PV of float leg  (positive for payer)
    StructField("swap_npv",   DoubleType()),   # Net NPV
    StructField("par_rate",   DoubleType()),   # Fair fixed rate that makes NPV = 0
    StructField("dv01",       DoubleType()),   # $ change per +1bp parallel SOFR shift
    StructField("duration",   DoubleType()),   # Modified duration in years
    StructField("pv01",       DoubleType()),   # DV01 per $1M notional ($/M per bp)
    StructField("error",      StringType()),
])


# ══════════════════════════════════════════════════════════════════════════════
# 4.  PRICING CORE  (pure Python — runs on PySpark worker nodes)
# ══════════════════════════════════════════════════════════════════════════════

_NAN = float("nan")
_NAN_ROW: Dict[str, Any] = dict(
    fixed_npv=_NAN, float_npv=_NAN, swap_npv=_NAN, par_rate=_NAN,
    dv01=_NAN, duration=_NAN, pv01=_NAN, error="",
)


def _price_one(row: Dict[str, Any], curve_df: pd.DataFrame) -> Dict[str, Any]:
    try:
        val_dt = row["valuation_date"]
        ql_val = ql.Date(val_dt.day, val_dt.month, val_dt.year)
        ql.Settings.instance().evaluationDate = ql_val

        sofr     = build_sofr_curve(ql_val, curve_df)
        calendar = ql.UnitedStates(ql.UnitedStates.GovernmentBond)
        dc_fixed = ql.Thirty360(ql.Thirty360.BondBasis)   # USD fixed: SA 30/360
        dc_float = ql.Actual360()                          # USD float: quarterly Act/360

        def to_ql(d):
            return ql.Date(d.day, d.month, d.year)

        eff = to_ql(row["effective_date"])
        mat = to_ql(row["maturity_date"])

        fixed_sch = ql.Schedule(
            eff, mat, ql.Period(ql.Semiannual), calendar,
            ql.ModifiedFollowing, ql.ModifiedFollowing,
            ql.DateGeneration.Forward, False,
        )
        float_sch = ql.Schedule(
            eff, mat, ql.Period(ql.Quarterly), calendar,
            ql.ModifiedFollowing, ql.ModifiedFollowing,
            ql.DateGeneration.Forward, False,
        )

        sofr_idx = ql.IborIndex(
            "SOFR3M", ql.Period(3, ql.Months), 2,
            ql.USDCurrency(), calendar, ql.ModifiedFollowing,
            False, dc_float, sofr,
        )

        # ── Provide historical fixings for any past reset dates ───────────────
        # For spot-starting swaps, the 1st float period resets 2 business days
        # before effective — which is before the valuation date.
        # QuantLib can't project rates before its reference date, so use the
        # short-end zero rate from our curve as the historical fixing.
        short_rate = float(curve_df["zero_rate"].iloc[0])   # 3M SOFR proxy
        eff_for_fix = eff
        for _ in range(36):
            fix_dt = calendar.advance(eff_for_fix, -2, ql.Days, ql.Preceding)
            if fix_dt > ql_val:
                break
            sofr_idx.addFixing(fix_dt, short_rate, True)
            eff_for_fix = calendar.advance(eff_for_fix, ql.Period(3, ql.Months),
                                           ql.ModifiedFollowing)

        stype    = (ql.VanillaSwap.Payer
                    if row["option_type"] == "PAYER"
                    else ql.VanillaSwap.Receiver)
        notional = float(row["notional"])
        fxd_rate = float(row["fixed_rate"])

        swap = ql.VanillaSwap(
            stype, notional, fixed_sch, fxd_rate, dc_fixed,
            float_sch, sofr_idx, 0.0, dc_float,
        )
        swap.setPricingEngine(ql.DiscountingSwapEngine(sofr))

        swap_npv  = swap.NPV()
        fixed_npv = swap.fixedLegNPV()
        float_npv = swap.floatingLegNPV()
        par_rate  = swap.fairRate()

        # ── DV01: parallel +1bp shift of the single SOFR curve ───────────────
        bumped            = curve_df.copy()
        bumped["zero_rate"] += 1e-4
        sofr_b   = build_sofr_curve(ql_val, bumped)
        sofr_idx_b = ql.IborIndex(
            "SOFR3M", ql.Period(3, ql.Months), 2,
            ql.USDCurrency(), calendar, ql.ModifiedFollowing,
            False, dc_float, sofr_b,
        )
        # Provide same historical fixings for bumped index
        eff_for_fix = eff
        for _ in range(36):
            fix_dt = calendar.advance(eff_for_fix, -2, ql.Days, ql.Preceding)
            if fix_dt > ql_val:
                break
            sofr_idx_b.addFixing(fix_dt, short_rate, True)
            eff_for_fix = calendar.advance(eff_for_fix, ql.Period(3, ql.Months),
                                           ql.ModifiedFollowing)
        swap_b = ql.VanillaSwap(
            stype, notional, fixed_sch, fxd_rate, dc_fixed,
            float_sch, sofr_idx_b, 0.0, dc_float,
        )
        swap_b.setPricingEngine(ql.DiscountingSwapEngine(sofr_b))
        dv01 = swap_b.NPV() - swap_npv

        # Modified duration = DV01 / (notional × 1bp)  → units: years
        # Positive for payer (short rate risk), negative for receiver (long rate risk)
        duration = dv01 / (notional * 1e-4)

        # PV01 = DV01 per $1M notional  (portfolio-level risk aggregation)
        pv01 = dv01 / notional * 1_000_000

        return dict(
            fixed_npv=fixed_npv, float_npv=float_npv, swap_npv=swap_npv,
            par_rate=par_rate, dv01=dv01, duration=duration, pv01=pv01,
            error="",
        )

    except Exception as exc:
        return {**_NAN_ROW, "error": str(exc)[:150]}


# ══════════════════════════════════════════════════════════════════════════════
# 5.  PANDAS UDF  (PySpark 4.x — pandas_udf with StructType return)
# ══════════════════════════════════════════════════════════════════════════════

_curve_broadcast = None   # set in run_pricing() before UDF creation


def make_price_udf():
    bc = _curve_broadcast

    @pandas_udf(RESULT_SCHEMA)
    def price_udf(
        trade_id:       pd.Series,
        option_type:    pd.Series,
        notional:       pd.Series,
        fixed_rate:     pd.Series,
        tenor_y:        pd.Series,
        effective_date: pd.Series,
        maturity_date:  pd.Series,
        valuation_date: pd.Series,
    ) -> pd.DataFrame:
        curve_df = bc.value
        results = [
            _price_one({
                "trade_id":       trade_id.iloc[i],
                "option_type":    option_type.iloc[i],
                "notional":       notional.iloc[i],
                "fixed_rate":     fixed_rate.iloc[i],
                "tenor_y":        int(tenor_y.iloc[i]),
                "effective_date": effective_date.iloc[i],
                "maturity_date":  maturity_date.iloc[i],
                "valuation_date": valuation_date.iloc[i],
            }, curve_df)
            for i in range(len(trade_id))
        ]
        return pd.DataFrame(results)

    return price_udf


# ══════════════════════════════════════════════════════════════════════════════
# 6.  SPARK RUNNER + ITERATIVE CONVERGENCE
# ══════════════════════════════════════════════════════════════════════════════

def run_pricing(n_trades: int = 100,
                tol: float = 1.0,
                max_iter: int = 5) -> pd.DataFrame:
    """
    Price n_trades iteratively until max|ΔNPV| < tol across consecutive runs.
    Proves: (1) results are deterministic, (2) parallel pricing is reproducible.
    """
    global _curve_broadcast

    spark = (
        SparkSession.builder
        .appName("VanillaSwapPricer")
        .master("local[*]")
        .config("spark.sql.execution.arrow.pyspark.enabled", "true")
        .config("spark.driver.extraJavaOptions",   "-Dlog4j2.rootLogger.level=OFF")
        .config("spark.executor.extraJavaOptions", "-Dlog4j2.rootLogger.level=OFF")
        .config("spark.ui.showConsoleProgress", "false")
        .getOrCreate()
    )
    spark.sparkContext.setLogLevel("OFF")
    n_cores = spark.sparkContext.defaultParallelism

    _real_stderr     = sys.stderr
    curve_df         = make_curve_df()
    _curve_broadcast = spark.sparkContext.broadcast(curve_df)

    trades_df = (
        spark.createDataFrame(make_trade_data(n_trades), schema=TRADE_SCHEMA)
             .repartition(n_cores)
    )
    price_udf = make_price_udf()

    def _execute() -> pd.DataFrame:
        return (
            trades_df
            .withColumn("r", price_udf(
                "trade_id", "option_type", "notional", "fixed_rate", "tenor_y",
                "effective_date", "maturity_date", "valuation_date",
            ))
            .select(
                "trade_id", "book", "counterparty", "option_type",
                "notional", "fixed_rate", "tenor_y",
                F.col("r.fixed_npv").alias("fixed_npv"),
                F.col("r.float_npv").alias("float_npv"),
                F.col("r.swap_npv").alias("swap_npv"),
                F.col("r.par_rate").alias("par_rate"),
                F.col("r.dv01").alias("dv01"),
                F.col("r.duration").alias("duration"),
                F.col("r.pv01").alias("pv01"),
                F.col("r.error").alias("error"),
            )
            .toPandas()
        )

    # ── Convergence loop ──────────────────────────────────────────────────────
    sep = "─" * 70
    print(sep)
    print(f"  Workers: {n_cores} cores  |  Trades: {n_trades}  |  Convergence tol: ${tol:.2f}")
    print(sep)

    prev_npv = None
    result   = None

    for it in range(1, max_iter + 1):
        sys.stderr = open(os.devnull, "w")
        t0      = time.time()
        result  = _execute()
        elapsed = time.time() - t0
        sys.stderr.close()
        sys.stderr = _real_stderr

        curr_npv = result["swap_npv"].values

        if prev_npv is not None:
            deltas    = np.abs(curr_npv - prev_npv)
            max_delta = float(np.nanmax(deltas))
            avg_delta = float(np.nanmean(deltas))
            print(
                f"  Iter {it}  {elapsed:5.1f}s  "
                f"max|ΔNPV| = ${max_delta:>10.4f}   "
                f"mean|ΔNPV| = ${avg_delta:>10.4f}",
                flush=True,
            )
            if max_delta < tol:
                print(f"  ✓ Converged after {it} iterations  (max|ΔNPV| < ${tol:.2f})\n",
                      flush=True)
                break
        else:
            print(f"  Iter {it}  {elapsed:5.1f}s  {len(result)} trades priced on "
                  f"{n_cores} parallel workers", flush=True)

        prev_npv = curr_npv.copy()

    sys.stderr = open(os.devnull, "w")
    spark.stop()
    sys.stderr.close()
    sys.stderr = _real_stderr

    return result

# 7.  PRINT TABLES  +  SANITY CHECKS  +  SAVE CSVs
# ══════════════════════════════════════════════════════════════════════════════

def print_results(df: pd.DataFrame, project_dir: str) -> None:
    sep = "=" * 97

    # ── Table 1: Identification / Rates / NPV ────────────────────────────────
    t1 = pd.DataFrame({
        "Trade":    df["trade_id"],
        "Book":     df["book"],
        "Cpty":     df["counterparty"],
        "Type":     df["option_type"],
        "Notl$M":   (df["notional"] / 1e6).map("{:>5.0f}".format),
        "Tenor":    df["tenor_y"].map("{}y".format),
        "FixRate":  df["fixed_rate"].map("{:.3%}".format),
        "ParRate":  df["par_rate"].map("{:.3%}".format),
        "SwapNPV":  df["swap_npv"].map("${:>13,.0f}".format),
    })
    print(f"\n{sep}")
    print("  VANILLA SWAP RESULTS (1/2)  —  Identification / Rates / NPV")
    print(sep)
    print(tabulate(t1, headers="keys", tablefmt="simple", showindex=False), flush=True)

    # ── Table 2: Leg NPVs / Risk ─────────────────────────────────────────────
    t2 = pd.DataFrame({
        "Trade":     df["trade_id"],
        "FixedNPV":  df["fixed_npv"].map("${:>13,.0f}".format),
        "FloatNPV":  df["float_npv"].map("${:>13,.0f}".format),
        "DV01($)":   df["dv01"].map("{:>9,.2f}".format),
        "Dur(yrs)":  df["duration"].map("{:>+8.2f}".format),
        "PV01($/M)": df["pv01"].map("{:>+8,.2f}".format),
    })
    print(f"\n{sep}")
    print("  VANILLA SWAP RESULTS (2/2)  —  Leg NPVs / Risk  "
          "[Duration: +ve=payer, -ve=receiver]")
    print(sep)
    print(tabulate(t2, headers="keys", tablefmt="simple", showindex=False), flush=True)

    # ── Portfolio Summary ─────────────────────────────────────────────────────
    pay = df[df["option_type"] == "PAYER"]
    rec = df[df["option_type"] == "RECEIVER"]
    err = df[df["error"].str.len() > 0] if "error" in df.columns else pd.DataFrame()

    print(f"\n{sep}")
    print("PORTFOLIO SUMMARY")
    print(f"  Trades            : {len(df):>6}  (Payer: {len(pay)}  |  Receiver: {len(rec)})")
    print(f"  Pricing errors    : {len(err):>6}")
    print(f"  Total Notional    : ${df['notional'].sum():>22,.0f}")
    print(f"  Total Swap NPV    : ${df['swap_npv'].sum():>22,.2f}")
    print(f"  Total DV01        : ${df['dv01'].sum():>22,.2f}")
    print(f"  Avg Par Rate      : {df['par_rate'].mean():>21.4%}")
    print(f"  Avg Fixed Rate    : {df['fixed_rate'].mean():>21.4%}")
    print(f"  Avg Duration      : {df['duration'].mean():>+20.2f} yrs")
    print(f"  Avg PV01 ($/M)    : {df['pv01'].mean():>+20,.2f}")
    print(f"  Payer NPV         : ${pay['swap_npv'].sum():>22,.2f}")
    print(f"  Receiver NPV      : ${rec['swap_npv'].sum():>22,.2f}")
    print(f"{sep}", flush=True)

    # ══════════════════════════════════════════════════════════════════════════
    # SANITY CHECKS
    # ══════════════════════════════════════════════════════════════════════════
    print("\nSANITY CHECKS")
    checks = []

    # 1. NPV reconciliation: fixed + float legs must sum to swap NPV
    recon = (df["fixed_npv"] + df["float_npv"] - df["swap_npv"]).abs()
    checks.append((
        "fixed_npv + float_npv = swap_npv",
        recon.max() < 0.01,
        f"max residual = ${recon.max():.5f}",
    ))

    # 2. DV01 sign: payer benefits from rising rates (+), receiver loses (-)
    sign_ok = (
        ((df["option_type"] == "PAYER")    & (df["dv01"] > 0)) |
        ((df["option_type"] == "RECEIVER") & (df["dv01"] < 0))
    ).all()
    checks.append((
        "DV01 sign  (payer > 0 = short rates,  receiver < 0 = long rates)",
        sign_ok,
        "all 100 trades",
    ))

    # 3. Par rates realistic for 2024 SOFR curve
    par_min, par_max = df["par_rate"].min(), df["par_rate"].max()
    checks.append((
        "Par rates in 3.8%–5.5%  (realistic 2024 SOFR)",
        (par_min > 0.038) and (par_max < 0.055),
        f"range: {par_min:.3%} – {par_max:.3%}",
    ))

    # 4. Duration / tenor ratio in sensible range
    #    At-par single-curve: ratio ≈ 0.97 (1y) down to 0.58 (30y)
    df["dur_ratio"] = df["duration"].abs() / df["tenor_y"]
    in_range = df["dur_ratio"].between(0.50, 1.05)
    pct_ok   = in_range.mean()
    checks.append((
        f"abs(Duration) within 50%–105% of tenor  ({pct_ok:.0%} of trades)",
        pct_ok >= 0.90,
        f"avg ratio = {df['dur_ratio'].mean():.3f}",
    ))

    # 5. DV01 magnitude vs annuity approximation
    #    Expected DV01 ≈ notional × duration (≈ 0.5–1.0 × tenor) × 0.0001
    df["dv01_approx"] = df["notional"] * df["tenor_y"] * 0.75 * 1e-4
    ratio_med = (df["dv01"].abs() / df["dv01_approx"]).median()
    checks.append((
        "DV01 magnitude  (median ratio vs annuity approx)",
        0.50 < ratio_med < 1.30,
        f"median ratio = {ratio_med:.3f}  (expected 0.50–1.30)",
    ))

    # 6. Near-par swaps: NPV should be tiny (< $2k per $10M notional)
    df["par_diff_bp"] = ((df["fixed_rate"] - df["par_rate"]) * 10_000).abs()
    near_par = df[df["par_diff_bp"] < 1.0]     # within 1bp of par
    if len(near_par):
        max_npv_np = near_par["swap_npv"].abs().max()
        checks.append((
            f"Near-par swaps NPV ≈ 0  ({len(near_par)} trades within 1bp of par)",
            max_npv_np < 5_000,
            f"max NPV = ${max_npv_np:,.0f}",
        ))

    # 7. At-par approximation: NPV ≈ (par - fixed) × notional × duration
    df["npv_approx"] = (
        (df["par_rate"] - df["fixed_rate"])
        * df["notional"]
        * df["duration"].abs()
        * 1e-4 / 1e-4    # duration already in years, no extra scaling needed
    )
    # Correct formula: NPV ≈ (par - fixed) × notional × annuity
    # annuity ≈ |duration| (in years)
    df["npv_approx2"] = (df["par_rate"] - df["fixed_rate"]) * df["notional"] * df["duration"].abs()
    # For payer: positive when par > fixed; for receiver: positive when fixed > par
    # Sign correction
    df["npv_sign"] = df["option_type"].map({"PAYER": 1.0, "RECEIVER": -1.0})
    df["npv_approx2"] *= df["npv_sign"]
    corr = df["swap_npv"].corr(df["npv_approx2"])
    checks.append((
        "NPV corr vs analytical approx  (par-fixed) × notional × duration",
        corr > 0.995,
        f"correlation = {corr:.5f}",
    ))

    # 8. No pricing errors
    n_err = len(err)
    checks.append((
        "No pricing errors",
        n_err == 0,
        f"{n_err} errors" if n_err else "clean — all 100 trades priced",
    ))

    for desc, passed, detail in checks:
        mark = "✓ PASS" if passed else "✗ FAIL"
        print(f"  {mark}  {desc}", flush=True)
        print(f"         [{detail}]", flush=True)

    if err is not None and len(err):
        print("\n  Pricing errors:")
        for _, row in err.iterrows():
            print(f"    {row['trade_id']}: {row['error']}")

    # ── Save CSVs ─────────────────────────────────────────────────────────────
    trade_csv = os.path.join(project_dir, "vanillaSwap.csv")
    port_csv  = os.path.join(project_dir, "vanillaSwapPortfolio.csv")

    # Drop helper columns before saving
    save_cols = [
        "trade_id", "book", "counterparty", "option_type", "notional",
        "fixed_rate", "tenor_y", "fixed_npv", "float_npv", "swap_npv",
        "par_rate", "dv01", "duration", "pv01", "error",
    ]
    df[save_cols].to_csv(trade_csv, index=False)

    summary = [
        ("trades",                   len(df)),
        ("payers",                   len(pay)),
        ("receivers",                len(rec)),
        ("pricing_errors",           n_err),
        ("total_notional",           df["notional"].sum()),
        ("total_swap_npv",           df["swap_npv"].sum()),
        ("total_dv01",               df["dv01"].sum()),
        ("avg_par_rate",             df["par_rate"].mean()),
        ("avg_fixed_rate",           df["fixed_rate"].mean()),
        ("avg_duration_yrs",         df["duration"].mean()),
        ("avg_pv01_per_million",     df["pv01"].mean()),
        ("payer_total_npv",          pay["swap_npv"].sum()),
        ("receiver_total_npv",       rec["swap_npv"].sum()),
        ("dv01_sign_check_pass",     sign_ok),
        ("npv_recon_max_residual",   recon.max()),
        ("duration_ratio_avg",       df["dur_ratio"].mean()),
        ("npv_analytical_corr",      corr),
    ]
    pd.DataFrame(summary, columns=["metric", "value"]).to_csv(port_csv, index=False)

    print(f"\n  Trade CSV      → {trade_csv}")
    print(f"  Portfolio CSV  → {port_csv}\n", flush=True)
